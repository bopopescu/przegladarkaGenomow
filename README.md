<ul>

        <b>pip:</b>(do instalacji django, psycopg2)<br>

    </li>
    <li>
        <b>django:</b><br>
        sudo pip install django==1.7
    </li>
    <li>
        <b>psycopg2:</b> <br>

    </li>
    <li>
        <b>scons:</b> <br>
        sudo apt-get install scons <br>
    </li>
    <li>
        <b>postgresql:</b> <br>
        sudo apt-get install postgresql <br>
    </li>
</ul>
